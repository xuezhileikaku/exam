<style type="text/css">
    /*默认边框 (Default border)*/
    .item-border .f-m-item-text
    {
        border:1px dotted gray;
        border-radius:4px;
        margin-left:3px;
    }

    /*鼠标停靠时的边框 (The mouse over at the border)*/
    .item-border .f-m-item:hover .f-m-item-text
    {
        text-decoration:none;
        border:1px solid Red;
        font-weight:bold;
    }

</style>

new fly.mini.Tree({
    //通过样式 "item-border" 控制节点边框
    //  Through the style "item-border" control node border
    css:'item-border',
    container: "treeContainer-css",
    checkMode: fly.mini.checkMode.multi,
    data: items_hasIcon
});
