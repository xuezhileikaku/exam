Checkbox view:
new fly.mini.Tree({
        checkMode: fly.mini.checkMode.multi,
        checkCascade: {
                check: { parent: true, children: true },
                uncheck: { parent: true, children: true }
        }
})
 Radiobox view:
new fly.mini.Tree({
        checkMode: fly.mini.checkMode.multi,
        checkStyle: fly.mini.checkStyle.radio,
        checkCascade: true
})
